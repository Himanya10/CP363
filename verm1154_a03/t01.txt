SELECT m.memberSurname, m.memberForename,
    (SELECT COUNT(*) FROM pub p WHERE p.pubMemberID = m.memberID AND p.pubPubType = 'B') AS books,
    (SELECT COUNT(*) FROM pub p WHERE p.pubMemberID = m.memberID AND p.pubPubType = 'A') AS articles,
    (SELECT COUNT(*) FROM pub p WHERE p.pubMemberID = m.memberID AND p.pubPubType = 'P') AS papers
FROM member m
ORDER BY m.memberSurname, m.memberForename
