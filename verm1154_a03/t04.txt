SELECT m.memberSurname, m.memberForename, b.broadDesc
FROM member m
JOIN memberBroad mb ON mb.memberBroadMemberID = m.memberID
JOIN broad b ON b.broadID = mb.memberBroadBroadId
WHERE NOT EXISTS (
    SELECT 1 FROM pub p WHERE p.pubMemberID = m.memberID
)
ORDER BY m.memberSurname, m.memberForename, b.broadDesc
